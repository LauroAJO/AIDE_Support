ROTAS ADICIONADAS (28 — todas novas, nenhuma rota existente removida ou reescrita):

MERCADO — Organizações
  GET    /api/market/organizations            (filtros ?type=&status=&search=&tags=; ordena relevance_score DESC, name ASC; inclui project_count e contact_count)
  POST   /api/market/organizations            (requer name; valida enum type; demais campos opcionais)
  GET    /api/market/organizations/:id        (org + projetos + contatos vinculados + oportunidades)
  PUT    /api/market/organizations/:id         (atualiza; valida type; atualiza updated_at)
  DELETE /api/market/organizations/:id         (só owner; bloqueia se há oportunidades ativas; ?confirm=true força)

MERCADO — Projetos
  GET    /api/market/projects                 (filtros ?organization_id=&type=&status=; inclui organization_name)
  POST   /api/market/projects                 (requer name)
  GET    /api/market/projects/:id             (projeto + organization_name)
  PUT    /api/market/projects/:id
  DELETE /api/market/projects/:id             (só owner)

MERCADO — Contatos Profissionais (extensão de network_people)
  GET    /api/market/contacts                 (network_people JOIN contact_professional; filtros ?outreach_status=&organization_id=; ordena next_action_date ASC, urgentes primeiro; inclui organization_name)
  GET    /api/market/contacts/:personId       (network_people + contact_professional + contact_org_links com org e projeto)
  POST   /api/market/contacts/:personId/professional   (cria/atualiza contact_professional; 404 se pessoa não existe em network_people)
  PUT    /api/market/contacts/:personId/professional   (atualiza; body.interaction é anexado ao interaction_history)
  POST   /api/market/contacts/link            (cria contact_org_links; requer person_id + organization_id)

MERCADO — Importação em massa
  POST   /api/market/import                   (owner OU assistant_fixed/fixed; INSERT OR IGNORE idempotente p/ organizations/contacts/links; retorna {imported,failed,errors}; grava import_log)

CARREIRA — Oportunidades
  GET    /api/career/opportunities            (filtros ?track=&status=&priority=; ordena por pipeline de status e deadline ASC; inclui organization_name e contact_name)
  POST   /api/career/opportunities            (requer title, type, track; created_by = user.id)
  GET    /api/career/opportunities/:id
  PUT    /api/career/opportunities/:id        (mudança de status é registrada em notes com timestamp)
  DELETE /api/career/opportunities/:id        (soft delete: status='closed')

CARREIRA — Documentos
  GET    /api/career/documents                (filtros ?type=&opportunity_id=)
  POST   /api/career/documents                (requer title)
